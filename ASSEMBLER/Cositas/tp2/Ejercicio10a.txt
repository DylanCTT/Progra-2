TIMER EQU 10H
PIC EQU 20H
HANDSHAKE EQU 40H

      ORG 1000H
PALABRA DB "INGENIERIA E INFORMATICA", 0

      ORG 2000H


      MOV BX, OFFSET PALABRA

      CMP BYTE PTR[BX], 0
      JZ ESPERAR

BUCLE:
      IN AL, HANDSHAKE + 1
      CMP AL, 1
      JZ BUCLE
      MOV AL, [BX]
      OUT HANDSHAKE, AL
      INC BX
      CMP BYTE PTR[BX], 0
      JNZ BUCLE

ESPERAR:
      JMP ESPERAR
      HLT
      END
